import React from 'react';
import { BrowserRouter as Router, Route, Routes, useNavigate } from 'react-router-dom';
import HomePage from './HomePage';
import ConfigPage from './ConfigPage';

function App() {
  return (
    <Router>
      <Routes>
        <Route path="/" element={<HomePage />} />
        <Route path="/config" element={<ConfigPage />} />
      </Routes>
    </Router>
  );
}

export default App;




import React from 'react';
import { useNavigate } from 'react-router-dom';
import { Button, Box, Typography } from '@mui/material';

function HomePage() {
  const navigate = useNavigate();

  const handleOpenConfig = () => {
    navigate('/config');
  };

  return (
    <Box
      sx={{
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        justifyContent: 'center',
        height: '100vh',
      }}
    >
      <Typography variant="h4" gutterBottom>Home Page</Typography>
      <Button variant="contained" color="primary" onClick={handleOpenConfig}>
        Open Configuration Settings
      </Button>
    </Box>
  );
}

export default HomePage;



import React from 'react';
import ConfigModal from './ConfigModal';
import { useNavigate } from 'react-router-dom';

function ConfigPage() {
  const navigate = useNavigate();

  const handleClose = () => {
    navigate('/');
  };

  return (
    <ConfigModal open={true} handleClose={handleClose} />
  );
}

export default ConfigPage;

import React, { useState } from 'react';
import { Modal, Box, Tabs, Tab, TextField, Button, InputAdornment, IconButton, Typography } from '@mui/material';
import CloseIcon from '@mui/icons-material/Close';
import HomeIcon from '@mui/icons-material/Home';

const TabPanel = ({ children, value, index }) => {
  return (
    <div role="tabpanel" hidden={value !== index}>
      {value === index && <Box sx={{ p: 3 }}>{children}</Box>}
    </div>
  );
};

const renderTextField = (label, value, disabled = false) => (
  <TextField
    fullWidth
    label={label}
    value={value}
    disabled={disabled}
    variant="outlined"
    margin="normal"
    InputProps={{
      startAdornment: <InputAdornment position="start">{label}</InputAdornment>,
    }}
  />
);

const ConfigModal = ({ open, handleClose }) => {
  const [tabValue, setTabValue] = useState(0);

  const handleTabChange = (event, newValue) => {
    setTabValue(newValue);
  };

  return (
    <Modal open={open} onClose={handleClose}>
      <Box
        sx={{
          position: 'absolute',
          top: '50%',
          left: '50%',
          transform: 'translate(-50%, -50%)',
          width: 600,
          bgcolor: 'background.paper',
          boxShadow: 24,
          p: 2,
          borderRadius: 2,
        }}
      >
        <Box sx={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', mb: 2 }}>
          <Typography variant="h6">Configuration Settings</Typography>
          <Box>
            <IconButton onClick={handleClose}>
              <CloseIcon />
            </IconButton>
            <IconButton onClick={handleClose}>
              <HomeIcon />
            </IconButton>
          </Box>
        </Box>

        <Tabs value={tabValue} onChange={handleTabChange} centered>
          <Tab label="Project Config" />
          <Tab label="User Config" />
        </Tabs>

        <TabPanel value={tabValue} index={0}>
          {renderTextField('Sandra Env', 'rates_qa')}
          {renderTextField('Sandra Path', '/Applications/qzratesqa/', true)}
          {renderTextField('Script Path', 'C:\\AutomationScripts\\UIAutomation\\')}
          {renderTextField('OpenFinDriver Path', 'C:\\AutomationScripts\\UIAutomation\\App\\')}
          {renderTextField('ChromeDriver Path', 'C:\\AutomationScripts\\UIAutomation\\App\\')}
        </TabPanel>

        <TabPanel value={tabValue} index={1}>
          {renderTextField('Sandra Env', 'rates_qa')}
          {renderTextField('Sandra Path', '/Applications/qzratesqa/', true)}
          {renderTextField('Script Path', '')}
          {renderTextField('OpenFinDriver Path', '')}
          {renderTextField('ChromeDriver Path', '')}
          <Button variant="contained" color="primary" fullWidth sx={{ mt: 2 }}>
            Save
          </Button>
        </TabPanel>
      </Box>
    </Modal>
  );
};

export default ConfigModal;
